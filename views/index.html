Cross-Origin Resource Sharing

Open the console to see the output of this code:


 const url = 'https://api.spotify.com/v1/users/jmperezperez';

 $.ajax(url).then(
   (data) => console.log(data),
   () => {
     console.error('Request is not allowed! Using the CORS hack instead:');
     $.ajax('/cors', { data: { url } }).then(
       data => console.log(data),
       (err) => console.error(err)
     );
   }
 );
